// Amaç: FAZ 1 düzeltmeleri:
//  - Enkoder atomik okuma (noInterrupts/interrupts)
//  - I2C timeout (Wire.setWireTimeout 5ms)
//  - Watchdog 2s (WDTO_2S)
//  - PID anti-windup (conditional integration + ±50.0f limit)
//  - PWM frekansı 31.25 kHz (Timer4 prescaler=1, phase-correct PWM)
//  - Zaman senkronizasyonu: payload.micros alanı doldurulur
//  - Struct packing: protocol.h ile uyumlu HhI
//  - CRC: protocol.h üzerinden üretilir

#include <Arduino.h>
#include <Wire.h>
#include <avr/wdt.h>

#include "protocol.h"

// Donanım bağımlı PWM ayarı (ATmega32U4 varsayımı - Timer4).
// Kullanılan PWM pinine uygun Timer ayarı yapınız (ör: Pin 6/13 -> Timer4).
static void setup_pwm_31k25Hz() {
  // Timer4: Phase Correct PWM, prescaler=1 -> ~31.37 kHz
  // Not: Gerçek uygulamada pin ve mod eşlemesini doğrulayın.
  // Clear prescaler bits, set to 1
  TCCR4B = (TCCR4B & 0b11111000) | 0x01;  // prescaler = 1
}

// Basit PID denetleyici
typedef struct {
  float kp;
  float ki;
  float kd;
  float integrator;
  float prev_error;
  float out_min;
  float out_max;
  float i_limit;  // ±50.0f anti-windup limiti
} pid_t;

static inline void pid_init(pid_t* pid, float kp, float ki, float kd, float out_min, float out_max) {
  pid->kp = kp;
  pid->ki = ki;
  pid->kd = kd;
  pid->integrator = 0.0f;
  pid->prev_error = 0.0f;
  pid->out_min = out_min;
  pid->out_max = out_max;
  pid->i_limit = 50.0f;  // FAZ 1: dar limit
}

static inline float clampf(float v, float lo, float hi) {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

static inline float pid_update(pid_t* pid, float setpoint, float measurement, float dt_s) {
  const float error = setpoint - measurement;

  // Türev (basit fark)
  const float deriv = (error - pid->prev_error) / (dt_s > 0.0f ? dt_s : 1e-3f);

  // Önce P ve D hesapla
  float p = pid->kp * error;
  float d = pid->kd * deriv;

  // Çıkışın saturasyonuna göre şartlı integral (anti-windup)
  // Koşul: Eğer önceki çıkış sınırda değilse ya da integral hatası sınırdan geri çekecek yönde ise entegre et.
  float tentative_i = pid->integrator + (pid->ki * error * dt_s);
  tentative_i = clampf(tentative_i, -pid->i_limit, pid->i_limit);

  float prelim_out = p + tentative_i + d;
  bool saturated = (prelim_out > pid->out_max) || (prelim_out < pid->out_min);

  if (!saturated) {
    pid->integrator = tentative_i;
  } else {
    // Eğer saturasyon var ve integral hatayı daha da kötüleştirecekse entegre etmeyin
    // Örn: üst saturasyonda ve error>0 ise veya alt saturasyonda ve error<0 ise atla
    if ((prelim_out > pid->out_max && error < 0.0f) ||
        (prelim_out < pid->out_min && error > 0.0f)) {
      pid->integrator = tentative_i;  // geri çeken yönde izin ver
    }
  }

  pid->prev_error = error;

  float out = p + pid->integrator + d;
  return clampf(out, pid->out_min, pid->out_max);
}

// Varsayılan enkoder sayacı (örnek)
volatile long g_encoder_count = 0;

// Enkoder kesmesi örneği (kendi donanımınıza göre uyarlayın)
void encoder_isr() {
  // Gerçek kodda pin durumuna göre +/− artış yapılır.
  g_encoder_count++;
}

static inline long read_encoder_atomic() {
  noInterrupts();          // FAZ 1: atomik okuma
  long v = g_encoder_count;
  interrupts();
  return v;
}

static pid_t g_pid;
static uint32_t last_micros = 0;

void setup() {
  // Seri başlat
  Serial.begin(115200);

  // I2C başlat ve timeout ayarla (5ms)
  Wire.begin();
#if ARDUINO >= 158 || defined(TWDR)  // derleyici/çekirdek uyumluluğu için basit koruma
  Wire.setWireTimeout(5000U, true);  // FAZ 1: 5ms timeout, hata durumunda TWI reset
#endif

  // PWM 31.25 kHz
  setup_pwm_31k25Hz();

  // PID
  pid_init(&g_pid, 0.8f, 0.2f, 0.01f, 0.0f, 255.0f);

  // Watchdog 2 saniye
  wdt_enable(WDTO_2S);  // FAZ 1: 2s WDT

  // Enkoder kesmesi bağlama örneği (uygun pine göre değiştirin)
  // attachInterrupt(digitalPinToInterrupt(2), encoder_isr, RISING);

  last_micros = micros();
}

void loop() {
  // Basit örnek döngü: enkoder oku, PID güncelle, çerçeve gönder
  const uint32_t now_us = micros();
  const float dt_s = (now_us - last_micros) * 1e-6f;
  last_micros = now_us;

  // Enkoder örneklemesi (atomik)
  const long enc = read_encoder_atomic();

  // Basit hedef ve ölçüm (temsilidir)
  const float set_rpm = 100.0f;
  const float meas_rpm = (float)enc * 0.1f;  // Örnekleme: uygulamaya göre hesaplanmalı

  // PID
  const float u = pid_update(&g_pid, set_rpm, meas_rpm, dt_s);

  // PWM çıkışı (0..255)
  const uint8_t pwm = (uint8_t)clampf(u, 0.0f, 255.0f);
  analogWrite(6, pwm);  // örnek pin

  // Telemetri paketini hazırla ve gönder
  proto_frame_t frame;
  frame.payload.msg_id = 0x0001;         // örnek mesaj id
  frame.payload.value = (int16_t)enc;    // HhI: 'h' alanına örnek değer
  frame.payload.micros = now_us;         // zaman senkronizasyonu için cihaz zamanı
  proto_build_frame(&frame);

  Serial.write((const uint8_t*)&frame, sizeof(frame));

  // Watchdog besle
  wdt_reset();

  // Basit döngü periyodu
  delay(5);
}
